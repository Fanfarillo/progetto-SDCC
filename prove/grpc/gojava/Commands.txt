1) INSTALL GO, JAVA, PROTOBUF-COMPILER AND MAVEN
sudo snap install go --classic
sudo apt install default-jdk
sudo apt install default-jre
sudo apt install -y protobuf-compiler
sudo apt install maven

2) MAKE SURE THAT PATH VARIABLE CONTAINS:
/go/bin:/.local/bin:${HOME}/go:${HOME}/go/bin:${HOME}/.bashrc
sudo nano /etc/environment
source /etc/environment

3) MAKE SURE THAT GOPATH VARIABLE CONTAINS:
/${HOME}/go
sudo nano /etc/environment
source /etc/environment

4) MAKE SURE THAT JAVA_HOME VARIABILE CONTAINS:
/usr/lib/jvm/{JAVA-JDK}
sudo nano /etc/environment
source /etc/environment

5) GENERATE go.mod AND go.sum FILES (in grpc/gojava directory)
go mod init gojava
go mod tidy

6) COMPILE PROTO FILE - CLIENT (in grpc/gojava directory)
protoc --proto_path=proto proto/*.proto --go_out=go/gen/
protoc --proto_path=proto proto/*.proto --go-grpc_out=go/gen/

7) COMPILE PROTO FILE - SERVER (in grpc/gojava/java directory)
mvn clean verify

8) EXECUTE SERVER (in grpc/gojava/java directory)
mvn exec:java -Dexec.mainClass="server.MicroServer"

9) EXECUTE CLIENT (in grpc/gojava directory)
go run ./go/client/MicroClient.go
